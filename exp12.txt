<h2>PHP Arrays</h2>

<?php
// 1. Indexed Array
$students = array("Amit", "Sneha", "Rahul", "Priya");

echo "<h3>Indexed Array - Student Names</h3>";

foreach ($students as $name) {
    echo $name . "<br>";
}


// 2. Associative Array
$marks = array(
    "Amit" => 85,
    "Sneha" => 90,
    "Rahul" => 78,
    "Priya" => 88
);

echo "<h3>Associative Array - Student Marks</h3>";

foreach ($marks as $name => $mark) {
    echo $name . " scored " . $mark . " marks<br>";
}
?>
